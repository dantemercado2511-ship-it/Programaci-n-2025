def quick_sort(L:list) -> list:
    if len(L) < 2:
        return L
    S = copy.copy(L)
    pivot = L[-1]
    div = 0
    for i in range(len(S)):
        if S[i] <= pivot:
            if i > div:
                swap(S,i,div)
            div += 1
    return quick_sort(S[:div-1]) + quick_sort(S[div-1:])
